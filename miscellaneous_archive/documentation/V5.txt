        function showNotification(message, type = 'info') {
            const notification = document.createElement('div');
            const colors = {
                success: 'bg-green-500/95 border-green-400',
                error: 'bg-red-500/95 border-red-400',
                info: 'bg-blue-500/95 border-blue-400',
                warning: 'bg-yellow-500/95 border-yellow-400'
            };
            
            notification.className = `fixed top-24 right-4 ${colors[type]} text-white px-5 py-3 rounded-lg shadow-2xl z-50 border flex items-center gap-3 font-medium`;
            
            const icons = {
                success: 'fa-circle-check',
                error: 'fa-circle-xmark',
                info: 'fa-circle-info',
                warning: 'fa-triangle-exclamation'
            };
            
            notification.innerHTML = `<i class="fa-solid ${icons[type]} text-lg"></i> ${message}`;
            document.body.appendChild(notification);
            
            setTimeout(() => {
                notification.style.opacity = '0';
                notification.style.transform = 'translateX(100px)';
                notification.style.transition = 'all 0.3s ease-out';
                setTimeout(() => notification.remove(), 300);
            }, 3500);
        }
